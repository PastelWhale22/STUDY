#include <string.h>

typedef struct {
    char name[50];
    int quantity;
} Product;


void sortProducts(Product products[], int size) {
    int i, j;
    Product temp;

    for (i = 0; i < size - 1; i++) {
        for (j = i + 1; j < size; j++) {
            if (strcmp(products[i].name, products[j].name) > 0) {
                temp = products[i];
                products[i] = products[j];
                products[j] = temp;
            }
        }
    }
}

void dataVegetableK(Product products[], int size) {
    sortProducts(products, size);

    printf("\n====목록====\n");
    printf("가나다 순으로 정렬된 농산물 목록:\n");

    for (int i = 0; i < size; i++) {
        printf("%s - 수량: %d\n", products[i].name, products[i].quantity);
    }
    printf("\n");
}

void addVegetable(Product products[], int* allVegetable) {
    printf("\n====입고====\n");

        printf("더 이상 농산물을 추가할 수 없습니다.\n");
        return;
    }

    printf("농산물 이름: ");

    printf("농산물 수량: ");
    scanf("%d", &products[*allVegetable].quantity);

}
